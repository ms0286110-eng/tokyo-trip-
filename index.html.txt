    <script>
        // 倒數計時
        const updateCD = () => {
            const diff = new Date('2026/04/08') - new Date();
            const days = Math.ceil(diff / (1000*60*60*24));
            document.getElementById('countdownDisplay').innerText = days > 0 ? `距離出發還有 ${days} 天` : '旅途愉快！';
        };
        updateCD();

        // 航班資料與切換
        const flights = {
            go: { f: 'TPE', t: 'NRT', timeF: '13:10', timeT: '17:35', info: 'GK 14 | 1航站', seat: '10B', lug: '7KG' },
            back: { f: 'NRT', t: 'TPE', timeF: '22:40', timeT: '01:30', info: 'GK 11 | 3航站', seat: '10F', lug: '25KG' }
        };
        const toggleFlight = (type) => {
            const f = flights[type];
            document.getElementById('goBtn').className = `tab-btn flex-1 py-2 rounded-lg ${type==='go'?'active':''}`;
            document.getElementById('backBtn').className = `tab-btn flex-1 py-2 rounded-lg ${type==='back'?'active':''}`;
            document.getElementById('flightContent').innerHTML = `
                <div class="flex justify-between items-center text-center">
                    <div><p class="text-3xl font-black text-blue-900">${f.f}</p><p class="text-[10px] text-gray-400">${f.timeF}</p></div>
                    <div class="flex-1 px-4 relative">
                        <div class="border-t-2 border-dashed border-gray-200 w-full mt-2"></div>
                        <i class="fa-solid fa-plane text-blue-500 absolute top-[-8px] left-1/2 -translate-x-1/2 bg-white px-2 text-xs"></i>
                        <p class="text-[9px] mt-1 text-blue-400 font-bold uppercase">${f.info}</p>
                    </div>
                    <div><p class="text-3xl font-black text-blue-900">${f.t}</p><p class="text-[10px] text-gray-400">${f.timeT}</p></div>
                </div>
            `;
        };
        toggleFlight('go');

        // 生成天氣
        const weatherHtml = [
            {d:'4/8', i:'fa-sun', v:'高'}, {d:'4/9', i:'fa-cloud', v:'中'}, {d:'4/10', i:'fa-cloud-showers-heavy', v:'低'},
            {d:'4/11', i:'fa-sun', v:'極高'}, {d:'4/12', i:'fa-sun', v:'高'}, {d:'4/13', i:'fa-cloud-sun', v:'中'}
        ].map(w => `
            <div class="bg-white p-3 rounded-2xl shadow-sm min-w-[85px] text-center border border-blue-50">
                <p class="text-[10px] text-gray-400 font-bold">${w.d}</p>
                <i class="fa-solid ${w.i} text-blue-400 text-lg my-2"></i>
                <p class="text-[9px] text-blue-600 bg-blue-50 rounded-full py-0.5">能見度: ${w.v}</p>
            </div>
        `).join('');
        document.getElementById('weatherScroll').innerHTML = weatherHtml;

        // 行程管理
        const addTrip = () => {
            const name = prompt("請輸入行程名稱 (例如：鎌倉半日遊)");
            if(!name) return;
            const div = document.createElement('div');
            div.className = "bg-white p-4 rounded-2xl shadow-sm border-l-4 border-blue-300 flex justify-between items-center";
            div.innerHTML = `<div><p class="text-[10px] text-gray-400 font-bold text-uppercase">自訂項目</p><h3 class="font-bold text-gray-800 text-sm">${name}</h3></div>
                             <div class="flex gap-3"><a href="https://www.google.com/maps/search/${encodeURIComponent(name)}" target="_blank" class="text-blue-400 text-sm"><i class="fa-solid fa-map-location-dot"></i></a>
                             <button onclick="this.parentElement.parentElement.remove()" class="text-red-300 text-sm"><i class="fa-solid fa-trash"></i></button></div>`;
            document.getElementById('tripContainer').appendChild(div);
        };

        // 清單與匯率
        const addTodo = () => {
            const val = document.getElementById('todoInput').value;
            if(!val) return;
            const li = document.createElement('li');
            li.className = "flex items-center justify-between p-2 bg-gray-50 rounded-lg";
            li.innerHTML = `<span onclick="this.classList.toggle('checked-item')">${val}</span><button onclick="this.parentElement.remove()" class="text-gray-300"><i class="fa-solid fa-xmark"></i></button>`;
            document.getElementById('todoList').appendChild(li);
            document.getElementById('todoInput').value = "";
        };

        const conv = (type) => {
            const r = 0.211;
            const j = document.getElementById('jpyV'), t = document.getElementById('twdV');
            if(type==='j') t.value = (j.value * r).toFixed(0);
            else j.value = (t.value / r).toFixed(0);
        };
    </script>
